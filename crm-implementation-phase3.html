⚙️ AsyncLocalStorage propagation
tenant-resolver.middleware.ts — vị trí quan trọng TypeScript
// TenantContext.run() bao quanh next() // → tất cả async continuations kế thừa context này // → L4 QueryInterceptor tự lấy tenantId mà không cần truyền tham số TenantContext.run( { tenantId: tenant.id, tenantTier: tenant.tier }, () => next() // Express chain → NestJS guards → controller → L4 );

Node.js AsyncLocalStorage đảm bảo: bất kỳ async operation nào được khởi tạo trong callback của run() đều kế thừa store — bao gồm Promise chain, event handler, setTimeout. Express's next() kick-starts NestJS processing trong đúng async context này.

TenantResolverMiddleware
src/gateway/middleware/tenant-resolver.middleware.ts (excerpt) TypeScript
@Injectable() export class TenantResolverMiddleware implements NestMiddleware { async use(req, _res, next) { // 1. Ưu tiên: X-Tenant-ID (UUID) → X-Tenant-Slug → subdomain const identifier = req.headers['x-tenant-id'] ?? req.headers['x-tenant-slug'] ?? extractSubdomain(req.headers['host'] ?? ''); if (!identifier) throw new BadRequestException('...'); // 2. Lookup DB (metadata pool — không phụ thuộc tenant scope) const tenant = await this.lookupTenant(identifier); if (!tenant) throw new NotFoundException(`Tenant not found: ${identifier}`); if (!tenant.isActive) throw new ForbiddenException('Tenant is inactive'); // 3. Gắn vào request để JwtAuthGuard cross-validate req.resolvedTenant = tenant; // 4. Set AsyncLocalStorage — L4 QueryInterceptor tự scope từ đây TenantContext.run( { tenantId: tenant.id, tenantTier: tenant.tier }, () => next() ); } }
JwtAuthGuard — Tenant Cross-Validation
src/gateway/guards/jwt-auth.guard.ts (excerpt) TypeScript
@Injectable() export class JwtAuthGuard extends AuthGuard('jwt') { handleRequest(err, user, info, context) { // 401: không có hoặc invalid JWT if (err || !user) throw new UnauthorizedException(err?.message ?? info?.message); // 403: JWT hợp lệ nhưng sai tenant // → ngăn token acme.app.com dùng cho beta.app.com const resolvedTenant = context.switchToHttp().getRequest().resolvedTenant; if (resolvedTenant && user.tenant_id !== resolvedTenant.id) { throw new ForbiddenException( `JWT tenant mismatch: token=${user.tenant_id}, request=${resolvedTenant.id}` ); } return user; // → req.user trong controller } }
JwtStrategy — JWKS + HS256 fallback
src/gateway/strategies/jwt.strategy.ts (excerpt) TypeScript
@Injectable() export class JwtStrategy extends PassportStrategy(Strategy, 'jwt') { constructor() { super({ jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(), algorithms: config.JWT_JWKS_URI ? ['RS256'] : ['HS256'], // Production: JWKS từ Keycloak (RS256, caching 10 phút) // Dev/Test: symmetric secret (HS256, min 32 chars) secretOrKeyProvider: config.JWT_JWKS_URI ? passportJwtSecret({ cache: true, jwksUri: config.JWT_JWKS_URI }) : config.JWT_SECRET_FALLBACK!, }); } async validate(payload) { // Verify sub + tenant_id có trong payload if (!payload.sub || !payload.tenant_id) throw new UnauthorizedException('JWT missing required claims'); return payload; // → req.user } }
